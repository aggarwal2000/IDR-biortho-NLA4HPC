1028 __global__ void CSR_SpMV(int N, const int* row_ptr_matrix, const int* col_ind_matrix, const DoubleComplex* val_matrix, const DoubleComplex* vec, DoubleComplex* vec_result)

1031  int gid = blockDim.x * blockIdx.x + threadIdx.x;
1032 
1033  int warp_index = (int)(gid / WARP_SIZE); //(32 is warp size)
1034 
1035  int row = warp_index;
1036 
1037  if (row < N) {
1038  int start_index_of_row = row_ptr_matrix[row]; //inclusive
1039  int end_index_of_row = row_ptr_matrix[row + 1]; //exclusive
1040 
1041  int id_within_warp = gid % 32;
1042 
1043 
1044  DoubleComplex temp = { 0,0 };
1045 
1046  for (int k = start_index_of_row + id_within_warp; k < end_index_of_row; k = k + 32)
1047  {
1048  temp = temp + val_matrix[k] * vec[col_ind_matrix[k]];
1049  }
1050 
1051  DoubleComplex val = temp;
1052 
1053  for (int offset = 16; offset > 0; offset /= 2)
1054  {
1055  val.x += __shfl_down_sync(FULL_MASK, val.x, offset);
1056  val.y += __shfl_down_sync(FULL_MASK, val.y, offset);
1057  }
1058 
1059  if (id_within_warp == 0)
1060  vec_result[row] = val;
1061 
1062  }
1063
